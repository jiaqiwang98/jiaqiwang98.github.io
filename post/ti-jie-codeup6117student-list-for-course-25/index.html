题解Codeup6117:Student List for Course (25)

2019-06-22

题目

http://codeup.cn/problem.php?cid=100000596&pid=1

这个专题的做题顺序应该是6117->6114......第一个直接暴力会超时，需要字符串Hash一下，6117具体的做法下回分解。

思路：

建立一个vector<string>型的数组v[course]，course是这门课的编号，数组里装选这门课的所有学生名字。然后根据题目要求输出课编号，选课人数即为v[i].size()，遍历输出每个名字。

坑点：Then output the students' names in alphabetical order. 输出的名字按字典序排列。没注意到这个你就会WA。

题解：

#include <iostream>
#include <cstring>
#include <vector>
#include <algorithm>
const int maxn = 2550;
using namespace std;
int main(){
    int n,k;
    cin>>n>>k;
    string name;
    vector<string> v[maxn];
    while(n--){
        int c,course;
        cin>>name;
        cin>>c;
        for (int i = 1; i <= c; ++i) {
            cin>>course;
            v[course].push_back(name);
        }
    }
    for (int j = 1; j <= k; ++j) {
        cout<<j<<" ";
        cout<<v[j].size()<<endl;
        sort(v[j].begin(),v[j].end());     //字典序排列容器中的名字字符串
        for (vector<string>::iterator it = v[j].begin();it != v[j].end(); it++) {
            cout<<*it<<endl;
        }
    }
    return 0;
}
