#include <iostream>
using namespace std;
bool visit[15];
int num[15];
int n;
void show(){
    for (int i = 1; i <= n; ++i) {
        cout<<num[i]<<" ";
    }
    cout<<endl;
}
void dfs(int step){
    if (step >  n){
        show();
    }
    for (int i = 1; i <= n; ++i) {
        if (visit[i] == 0){
            num[step] = i;
            visit[i] = 1;
            dfs(step + 1);
            visit[i] = 0;
        }
    }
}
int main(){
    cin>>n;
    dfs(1);
    return 0;
}

#include <iostream>
using namespace std;
int col[100] = {0},d1[100] = {0},d2[100] = {0};
int vis[100] = {0};
int n;
int cnt = 0;
int ans = 0;
//d1左下到右上(i+j),d2左上到右下(i-j+n)
//i为行,j为列
void print(){
    if(ans < 3){
        for (int i = 1; i <= n; ++i) {
            cout<<vis[i]<<" ";
        }
        cout<<endl;
    }

    cnt++;
    //return;
}
void dfs(int step){
    if (step > n){
        print();
        ans ++;
        return;
    }
    for (int i = 1; i <= n; ++i) {
        if (col[i] == 0 && d1[step + i] == 0 && d2[step - i + n] == 0){
            vis[step] = i;
            col[i] = 1;
            d1[step + i] = 1;
            d2[step - i + n] = 1;
            //cnt++;
            dfs(step + 1);
            //cnt--;
            col[i] = 0;
            d1[step + i] = 0;
            d2[step - i + n] = 0;
        }
    }
}
int main(){
    cin>>n;
    dfs(1);
    cout<<cnt;
    return 0;
}
