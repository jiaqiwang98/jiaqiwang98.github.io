题解PAT B1023:组个最小数

· 2019-06-19 · # 刷题笔记

题目

https://pintia.cn/problem-sets/994805260223102976/problems/994805298269634560

我的思路

没什么好说的，贪心基本题，0不能在第一位

题解

#include <iostream>
using namespace std;
int main(){
    int a[10];
    for (int i = 0; i < 10; ++i) {
        cin>>a[i];
    }
    for (int k = 1; k < 10; ++k) {
        if (a[k] != 0){
            cout<<k;
            a[k] --;
            break;
        }
    }
    for (int j = 0; j < 10; ) {
        if (a[j] != 0){
            cout<<j;
            a[j] --;
        }
        if(a[j] == 0){
            j++;
        }
    }
    return 0;
}
