题解洛谷P3370:【模板】字符串哈希

2019-06-19

正题之前先看一下乞丐版的字符串Hash：

问题：
给出N个字符串（由恰好三位大写字母组成），再给出M个查询字符串，问每个查询字符串在N个字符串中出现的次数？

思路就是将字符串每一位看成26进制数的每一位（因为题目只有大写，如果有小写的话就是52进制，以此类推，包含所有ASCII码的字符串就是127进制），然后进行换算得出唯一的Hash值

题解

#include <iostream>
#include <cstring>
using namespace std;
int hashFunc(char s[],int len){
    int id = 0;
    for (int i = 0; i < len; ++i) {
        id = id * 26 + (s[i] - 'A');
    }
    return id;
}
int main(){
    int n,m;
    cin>>n;
    char s[10];
    int hashTable[26 * 26 * 26 + 10];
    memset(hashTable,0,sizeof(hashTable));
    for (int i = 0; i < n; ++i) {
        scanf("%s",s);
        hashTable[hashFunc(s,3)] ++;
    }
    cin>>m;
    int t;
    while(m--){
        scanf("%s",s);
        t = hashTable[hashFunc(s,3)];
        cout<<t<<endl;
    }
    return 0;
}

#define mod 19260817
using namespace std;
long long hashFunc(char s[],int len){
    int id = 0;
    for (int i = 0; i < len; ++i) {
        if (s[i] > 'A' && s[i] < 'Z'){
            id = (id * b + (s[i] - 'A')) % mod;
        }else if(s[i] > 'a' && s[i] < 'z'){
            id = (id * b + (s[i] - 'a') + 26) % mod;
        }else if (s[i] > '0' && s[i] < '9'){
            id = (id * b + (s[i] - '0') + 52) % mod;
        }
    }
    return id;
}

思路：unsigned long long类型具有自动溢出取模功能，如果unsigned long long溢出会自动模2^64，这样就不需要我们寻找大素数取模了。（好吧我承认我也是看题解区才知道的...）

#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
using namespace std;
typedef unsigned long long ull;//宏定义
#define b 131
ull hashFunc(char s[], int len){
    ull id = 0;
    for (int i = 0; i < len; ++i) {
        id = id * b + (ull)s[i];
    }
    return id;
}
int main(){
    char s[10500];
    ull hashTable[10050];
    memset(hashTable,0, sizeof(hashTable));
    int n,cnt = 1;
    cin>>n;
    for (int i = 0; i < n; ++i) {
        scanf("%s",s);
        hashTable[i] = hashFunc(s,strlen(s));
    }
    sort(hashTable,hashTable + n);
    for (int j = 1; j < n; ++j) {
        if (hashTable[j] != hashTable[j - 1]){
            cnt++;
        }
    }
    cout<<cnt<<endl;
    return 0;
}
