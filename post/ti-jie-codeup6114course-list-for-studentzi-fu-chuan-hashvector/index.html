题解Codeup6114:Course List for Student(字符串Hash+vector)

2019-06-22

问题：

http://codeup.cn/problem.php?cid=100000596&pid=0

思路：

这道题相当于Codeup6117(http://codeup.cn/problem.php?cid=100000596&pid=1)的流程反过来，所以一开始打算用Hash+vector存，但是想了想这道题在STL这一章应该用不到Hash，就写出了如下的暴力代码。

结果:时间超限14%

#include <iostream>
#include <cstring>
#include <vector>
#include <algorithm>
const int maxn = 2550;
const int maxs = 40005;
using namespace std;
int main(){
    int n,k;
    cin>>n>>k;
    vector<string> v[maxn];
    vector<int > na;
    for (int j = 1; j <= k; ++j) {
        string name;
        int index,nums;
        cin>>index>>nums;
        for (int i = 1; i <= nums; ++i) {
            cin>>name;
            v[index].push_back(name);
        }
    }
    string q;
    while(cin>>q){
        cout<<q<<" ";
        for (int i = 1; i <= k; ++i) {
            for (auto it = v[i].begin(); it != v[i].end(); it++) {
                if(*it == q){
                    na.push_back(i);
                }
            }
        }
        cout<<na.size()<<" ";
        for(auto it2 = na.begin(); it2 != na.end(); it2++){

            cout<<*(it2)<<" ";
        }
        cout<<endl;
        na.clear();
    }
    return 0;
}

题解：

以下是Hash+vector的AC代码

AC

需要注意！输出容器内课程编号应排序为升序，原题**and finally the indices of the courses in increasing order.** 否则会“答案错误57%”

#include <iostream>
#include <vector>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxn = 26 * 26 * 26 * 10 + 10;
int hashTable(char s[]){		
//满足最后一位是数字，可以直接用《算法笔记》中的Hash模版
    int id = 0;
    for (int i = 0; i < strlen(s) - 1; ++i) {
        id = id * 26 + (s[i] - 'A');
    }
    id = id * 10 + (s[strlen(s) - 1] - '0');
    return id;
}
int main(){
    int n,k;
    char name[5];
    scanf("%d%d",&n,&k);	//scanf尽可能替换cin
    vector<int > v[maxn];
    for (int i = 1; i <= k; ++i) {
        int c,c_nums;
        scanf("%d%d" ,&c ,&c_nums);
        for (int j = 1; j <= c_nums; ++j) {
            scanf("%s",name);
            v[hashTable(name)].push_back(c);
        }
    }
    while(scanf("%s",name)!=EOF){
        printf("%s ",name);           //printf尽可能替换cout
        printf("%d ",v[hashTable(name)].size());
        sort(v[hashTable(name)].begin(),v[hashTable(name)].end());   
        //样例输出为升序，如果不加sort会“答案错误57%”
        for (vector<int>::iterator it = v[hashTable(name)].begin();
        it != v[hashTable(name)].end();it++) {
            cout<<*it<<" ";
        }
        cout<<endl;
    }
    return 0;
}
